\section{Graphic output ({\tt wfgout})}
Indicate graphics of spatial profile and time evolution.  One graphic
control command separated by a space corresponds to one figure.
Multiple figures are drawn on a page by writing a number of commands in a
line.

\subsection{Example of input data}
\begin{itemize}
\item[]
Example 1: \t{EXR EYI EZA}
\begin{itemize}
\item
The following three figures are drawn on a page.
\item[]
Contour of the real part of the $x$ component of the wave electric
filed $E_x$
\item[]
Contour of the imaginary part of the $x$ component of the wave electric
filed $E_x$
\item[]
Contour of the absolute value of the $z$ component of the wave electric
filed $E_z$
\end{itemize}

\item[]
Example 2: \t{EYX0.0 PTEC PNIY0.0}
\begin{itemize}
\item[]
$x$ dependence of the $y$ component of the wave electric field $E_y$ at $y=0.0$
\item
Contour of electron temperature
\item
$y$ dependence of the ion density at $x=0.0$
\end{itemize}

\item[]
Example.3: \t{TN1}
\begin{itemize}
\item[]
Time dependence of the electron density (solid line: average, dashed line:
maximum, broken line: minimum)
\end{itemize}

\item[]
Example 4: \t{5}
\begin{itemize}
\item[]
Profiles of electrostatic potential, density, temperature
\item
Character strings defined by \t{KGINX(5)} in \t{WFINIT}.
\end{itemize}
\end{itemize}


\subsection{Definition of graphic control command}
\begin{tabular}{ll}
\t{E}\Xa\Xb & spatial profile of wave E \\
\t{B}\Xa\Xb & spatial profile of wave B \\
\t{A}\Xa\Xb & spatial profile of wave potential \\
\t{P}\Xa\Xb\Xc & spatial profile of plasma quantities \\
\t{T}\Xa\Xb & time evolution of plasma quantities \\
$n$            & combination of pre-defined string in (\t{KGINX}) \\
\t{V}$n$       & combination of pre-defined string in (\t{KGINV}) \\
\t{G}$n$       & type of graphic output \\
\t{L}$n$       & number of figures in horizontal direction on a page \\
\t{X}          & exit
\end{tabular}
